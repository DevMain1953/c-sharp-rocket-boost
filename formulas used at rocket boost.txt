2D_rotation_matrix:
[[cosine(angle), -sinus(angle), 0]]
[[ sinus(angle), cosine(angle), 0]]
[[            0,             0, 1]]

yaw angle in radians = get arctangent of(2D_rotation_matrix[2, 1], 2D_rotation_matrix[1, 1]);
yaw angle in degress = (yaw angle in radians) / (180 / PI);
